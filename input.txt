{
    int i ; int j ; float v ; float x ; float[100] a ;
    while( true ) {
        do i = i+1 ; while( a[i] < v) ;
        do j = j-1 ; while( a[j] > v) ;
        float h;
	float k;
	if( i >= j ) break;
        x = a[i] ; a[i] = a[j] ; a[j] = x ;
    }

    a = ((i + j[h][k]) * k) ;
}
